void WalkableSurroundRule::Evaluate(const Tower& aTower, FloorViolationData& aViolationData, int aFloorIndex)
{
	const FloorData& currentFloorData = aTower[aFloorIndex];

	if (currentFloorData.walkableBlocksMetaData.empty())
	{
		return;
	}

	for (const BlockCoordinate& walkableCoordinate : currentFloorData.walkableBlocksMetaData)
	{
		for (int i = 0; i < static_cast<int>(Direction::Count); ++i)
		{
			const BlockCoordinate adjacentCoordinate = GetAdjacentCoordinate(walkableCoordinate, static_cast<Direction>(i));

			if (!IsCoordinateWithinBounds(adjacentCoordinate))
			{
				myIsValid = false;
				aViolationData.hasFloorViolation = true;
				break;
			}

			const BlockData& adjacentBlockData = currentFloorData.blockData[adjacentCoordinate.row][adjacentCoordinate.column];

			if (adjacentBlockData.blockType == BlockType::Void)
			{
				if (aFloorIndex == 0)
				{
					myIsValid = false;
					aViolationData.hasFloorViolation = true;
					break;
				}

				const FloorData& belowFloorData = aTower[aFloorIndex - 1];
				const BlockData& adjacentBelowBlockData = belowFloorData.blockData[adjacentCoordinate.row][adjacentCoordinate.column];

				if (adjacentBelowBlockData.blockType == BlockType::Stairs)
				{
					continue;
				}

				myIsValid = false;
				aViolationData.hasFloorViolation = true;
			}
		}
	}


}
